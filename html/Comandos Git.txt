
git config --global user.name "<nombre>"
git config --global user.email "<correo>"
git config --list 

git init - Inicia un repositorio de manera local
git remote add origin <url> - Enlaza el repositorio local con el remoto
git remote -v - Mustra la url del repositorio remoto
git status [--short] - Muesta el estado de los archivo
git add . - Adiciona todos los archivo a modo stage
git commit -m <"texto"> - Adiciona de forma permanente los cambios
git push - Sube los cambios al repositorio remoto
